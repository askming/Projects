model{	# parameters for QR       k1 <- (1-2*qt)/(qt*(1-qt))       k2 <- 2/(qt*(1-qt))       for (i in 1:I){	# longitudinal process, QR mixed model		for (j in 1:J){			er[i,j] ~ dexp(sigma)			mu[i,j] <- u[i,1]*Z[i,1] + u[i,2]*t[j] + inprod(X[i,], beta[]) + delta[1]*H[i,1] + delta[2]*H[i,1]*t[j] + k1*er[i,j]			prec[i,j] <- sigma/(k2*er[i,j])			y[i,j] ~ dnorm(mu[i,j],prec[i,j])		} # end of j loop	# survival process	h[i] <- exp(inprod(gamma, W[i,]) + alpha1*(delta[1]*H[i,1] + delta[2]*H[i,1]*t[j]) + alpha2*(u[i,1]*Z[i,1] + u[i,2]*t[j])) # hazard function	S[i] <- exp(- (h[i]-exp(alpha1*delta[1]*H[i,1]+alpha2*u[i,1]+inprod(gamma, W[i,])))/(alpha2*u[i,2]+alpha1*delta[2]*H[i,2])) # survival function	L[i] <- pow(h[i], event[i]) * pow(S[i], 1-event[i]) # likelihood for subject i	zeros[i] <- 0	phi[i] <- -log(L[i])	zeros[i] ~ dpois(phi[i])	# prior for random effects ~ mvn(mean, Sigma), use Cholesky decomposition for variance	u[i,1] <- z[i,1]	u[i,2] <- w21*z[i,1] + w22*z[i,2]	z[i, 1:2] ~ dmnorm(zero[], identity[,])		w22 ~ dunif(0, 10)	w21 ~ dnorm(0, 0.01)		Sigma[1,1]<-1	Sigma[1,2]<-w21	Sigma[2,1]<-w21	Sigma[2,2]<-w21*w21+w22*w22			} # end of i loop			# priors for other parameters	beta1 ~ dnorm(0, 0.001)	beta2 ~ dnorm(0, 0.001)	delta1 ~ dnorm(0, 0.001)	delta2 ~ dnorm(0, 0.001)	gamma1 ~ dnorm(0, 0.001)	gamma2 ~ dnorm(0, 0.001)	sigma ~ dgamma(0.001,0.001)}
